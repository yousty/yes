def loader
  @loader ||= begin
    loader = Zeitwerk::Loader.new
    loader.tag = 'yes-core'
    loader.push_dir(File.expand_path('..', __dir__))
    loader.ignore("#{__dir__}/core/version.rb")
    loader.collapse("#{__dir__}/core/models")
    loader.setup
    loader
  end
end
