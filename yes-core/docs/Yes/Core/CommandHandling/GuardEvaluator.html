def method_missing(method_name, *, &)
  if aggregate.respond_to?(method_name)
    result = aggregate.public_send(method_name, *, &)
    track_external_aggregate(method_name, result) if aggregate_attribute?(method_name)
    result
  else
    super
  end
end
