def error_details
  return {} unless error

  {
    message: error.message,
    type: error.message&.underscore&.tr(' ', '_'),
    extra: error.extra
  }.compact
end

def to_notification
  error = success? ? {} : { error_details: }
  {
    type:,
    batch_id:,
    payload:,
    metadata:,
    command: cmd.class.name,
    id: cmd.command_id,
    transaction: transaction.to_h
  }.merge(error)
end
